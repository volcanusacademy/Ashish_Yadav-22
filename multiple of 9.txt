// Online C compiler to run C program online
#include <stdio.h>

int main() {
    // declared variable
    int start,end;
    // taking start and end range from user
    printf("Enter starting number:");
    scanf("%d",&start);
    printf("Enter ending number:");
    scanf("%d",&end);
    
    // forloop run between range of user
    for(int i =start;i<=end;i++){
        // condition apply
        if(i%9==0){
            printf("%d\n",i);
            break;
        }
        else{
            printf("%d\n",i);
        }
    }
    return 0;
}
