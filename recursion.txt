#include <stdio.h>

int facTorial(int num){
    if(num>=1){
        return num*facTorial(num-1);
    }
    else{
        return 1;
    }
}
int main() {
    int num;
    printf("Enter number:");
    scanf("%d",&num);
    printf("factorial of %d is:%d",num,facTorial(num));
    return 0;
}
