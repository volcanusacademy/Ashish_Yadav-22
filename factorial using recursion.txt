// Online C compiler to run C program online
#include <stdio.h>
int n;
int factorial(int n){
    if(n>1){
        return n*factorial(n-1);
    }
    else{
        return 1;
    }
}
int main() {
    int n;
    printf("Enter number:");
    scanf("%d",&n);
    printf("Factorial is:%d",factorial(n));
    return 0;
}
