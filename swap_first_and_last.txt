#include <stdio.h>

int num,temp,count=0;
int arr[5],arrtemp[5];

int CouNt(){
    //count length of number function
    int i=num;
    while(i!=0){
        i = i/10;
        count++;
    }
    return count;
}

int Swap(int num){
    // number put in arr[] example = num=12345 then put in array 54321
    for(int i=0;i<=4;i++){
        arr[i] = num%10;
        num = num/10;
    }
    // arr[value] put in arrtemp[value] because set in arr in orignal form example 12345
    for(int i=4;i>=0;i--){
        arrtemp[i]=arr[i];
    }
    // swap first and last 
    temp = arrtemp[0];
    arrtemp[0]=arrtemp[4];
    arrtemp[4]=temp;
    // print swap array
    for(int i=4;i>=0;i--){
        printf("%d",arrtemp[i]);
    }
}

int main() {
    printf("Enter five digit number:");
    scanf("%d",&num);
    
    if(CouNt()==5){
        Swap(num);
    }
    else{
        printf("Please Enter only five digit number!\n");
    }
    return 0;
}
