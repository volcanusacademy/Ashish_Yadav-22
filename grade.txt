#include <stdio.h>
float sum=0;

    float marks[5];
    for(int i=0;i<5;i++){
        if(i==0){
            printf("Enter Hindi marks:");
        }
        if(i==1){
            printf("Enter English marks:");   
        }
        if(i==2){
            printf("Enter Maths marks:");   
        }
        if(i==3){
            printf("Enter Science marks:");   
        }
        if(i==4){
            printf("Enter Physics marks:");   
        }
        scanf("%f",&marks[i]);
        sum = sum+marks[i];
    }
    return sum;
}

float Per(){

    return sum;
}
