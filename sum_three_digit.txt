#include <stdio.h>
#include <math.h>

int num,rem,sum=0,count=0;

int CouNt(){
    int i=num;
    while(i!=0){
        i = i/10;
        count++;
    }
    return count;
}

int Sum(int num){
    while(num!=0){
        rem = num%10;
        sum += rem;
        num = num/10;
    }
    printf("Sum of three digit is %d\n",sum);
}

int main() {
    printf("Enter three digit number:");
    scanf("%d",&num);
    
    if(CouNt()==3){
        Sum(num);
    }
    else{
        printf("Please Enter only three digit number!");
    }
    
    
    return 0;
}
