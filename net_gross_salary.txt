#include <stdio.h>
// ta 10
// pf 7.8
// da=500
//  gs=basic+da+ta; ns=gs-pf;
float Basic_Salary;
float TA,PF,DA=500;
float gross_salary,net_salary;

float Travel_Allowance(float Basic_Salary){
    TA = (Basic_Salary*10)/100;
}
float Provident_Fund(float Basic_Salary){
    PF = (Basic_Salary*7.8)/100;
}
float CalCulate_Salary(){
    gross_salary = Basic_Salary+DA+TA;
    printf("your gross salary is :%.2frs\n",gross_salary);
    net_salary = gross_salary-PF;
    printf("your net salary is :%.2frs\n",net_salary);
}

int main() {
    printf("Enter your basic salary\n");
    scanf("%f",&Basic_Salary);
    Travel_Allowance(Basic_Salary);
    Provident_Fund(Basic_Salary);
    CalCulate_Salary();
    return 0;
}
