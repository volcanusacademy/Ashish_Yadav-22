#include <stdio.h>
#include <math.h>
float simPLe_IntRest(float p,float r,float t){
    float SI = p*r*t/100;
    printf("The simple interest is %.2f\n",SI);
    printf("Total amount after interest %.2f\n",p+SI);
}

float comPound_IntRest(float p,float r,float t){
    float a = p * pow((1 + r / 100), t);
    float CI = a-p;
    printf("The Compound interest is %.2f\n",CI);
    printf("Total amount after interest %.2f\n",p+CI);
}

int main() {
    float p,r,t;
    printf("Enter the principle:",p);
    scanf("%f",&p);
    printf("Enter the rate:",r);
    scanf("%f",&r);
    printf("Enter the time:",t);
    scanf("%f",&t);
    
    if(p>500){
        comPound_IntRest(p,r,t);
    }
    else{
        simPLe_IntRest(p,r,t);
    }
    return 0;
}
