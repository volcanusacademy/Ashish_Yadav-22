#include <stdio.h>
int num,count=0,rem,sum=0;
int rem1,sum1=0;
int Count(int num){
    int i=num;
    while(i!=0){
        i = i/10;
        count++;
    }
    return count;
}

int main() {
    printf("Enter Four digit number\n");
    scanf("%d",&num);
    
    if(Count(num)==4){
        while(num!=0){
            rem = num%10;
            sum += rem;
            num = num/10;
        }
        if(sum>9){
            while(sum!=0){
                rem1 = sum%10;
                sum1 = sum1+rem1;
                sum= sum/10;
            }
            printf("sum of four digit is %d\n",sum1);
        }
        else{
            printf("Sum of four digit is: %d\n",sum);
        }
    }
    else{
        printf("Please enter only Four digit number\n");
    }

    return 0;
}
