#include <stdio.h>
int num,count=0,rem;

int Count(int num){
    int i=num;
    while(i!=0){
        i = i/10;
        count++;
    }
    return count;
}

int main() {
    printf("Enter three digit number\n");
    scanf("%d",&num);
    
    if(Count(num)==3){
        rem = num%10;
        if(rem==5){
            printf("%d",num);
        }
        else if(rem<5){
            num = (num/10)*10;
            printf("%d",num);
        }
        else{
            num = (num/10)*10+10;
            printf("%d",num);
        }
    }
    else{
        printf("Please enter only three digit number\n");
    }

    return 0;
}
