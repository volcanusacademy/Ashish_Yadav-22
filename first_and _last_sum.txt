#include <stdio.h>

int num,sum,count=0;
int arr[6];

int CouNt(){
    int i=num;
    while(i!=0){
        i = i/10;
        count++;
    }
    return count;
}

int Sum(int num){
    for(int i=0;i<=5;i++){
        arr[i] = num%10;
        num = num/10;
    }
    sum = arr[0]+arr[5];
    printf("Sum first and last of six digit is %d\n",sum);
}


int main() {
    printf("Enter six digit number:");
    scanf("%d",&num);
    
    if(CouNt()==6){
        Sum(num);
    }
    else{
        printf("Please Enter only Six digit number!\n");
    }
    return 0;
}
